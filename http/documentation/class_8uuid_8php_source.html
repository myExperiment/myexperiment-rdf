00210      * Generate UUID version 1 (time based)
00211      */
00212     static private function generateTime($ns, $node) {
00213         $uuid = self::$m_uuid_field;
00214 
00215         /*
00216          * Get current time in 100 ns intervals. The magic value
00217          * is the offset between UNIX epoch and the UUID UTC
00218          * time base October 15, 1582.
00219          */
00220         $tp = gettimeofday();
00221         $time = ($tp['sec'] * 10000000) + ($tp['usec'] * 10) +
00222             0x01B21DD213814000;
00223 
00224         $uuid['time_low'] = $time & 0xffffffff;
00225         /* Work around PHP 32-bit bit-operation limits */
00226         $high = intval($time / 0xffffffff);
00227         $uuid['time_mid'] = $high & 0xffff;
00228         $uuid['time_hi'] = (($high >> 16) & 0xfff) | (self::UUID_TIME << 12);
00229         
00230         /*
00231          * We don't support saved state information and generate
00232          * a random clock sequence each time.
00233          */
00234         $uuid['clock_seq_hi'] = 0x80 | mt_rand(0, 64);
00235         $uuid['clock_seq_low'] = mt_rand(0, 255);
00236 
00237         /*
00238          * Node should be set to the 48-bit IEEE node identifier, but
00239          * we leave it for the user to supply the node.
00240          */
00241         for ($i = 0; $i < 6; $i++)
00242             $uuid['node'][$i] = ord(substr($node, $i, 1));
00243 
00244         return ($uuid);
00245     }
